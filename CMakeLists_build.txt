cmake_minimum_required(VERSION 3.16)
project(VideoSketchEffectBuild VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Optimization flags for release
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -march=native -mtune=native -DNDEBUG")

# Find required packages
find_package(OpenCV REQUIRED COMPONENTS core imgproc videoio imgcodecs)
find_package(Threads REQUIRED)

# Create executable with _build suffix
add_executable(sketch_video_build sketch_video_build.cpp)

# Link libraries
target_link_libraries(sketch_video_build 
    ${OpenCV_LIBS} 
    Threads::Threads
)

# Enable OpenMP if available for additional parallelization
find_package(OpenMP)
if(OpenMP_CXX_FOUND)
    target_link_libraries(sketch_video_build OpenMP::OpenMP_CXX)
    message(STATUS "OpenMP found - enabling parallel optimizations")
endif()

# Display build information
message(STATUS "Building Video Sketch Effect Build v${PROJECT_VERSION}")
message(STATUS "OpenCV version: ${OpenCV_VERSION}")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
